THREE = {
  const THREE = window.THREE = await require("three@0.145.0/build/three.min.js");
  await require("three@0.145.0/examples/js/controls/OrbitControls.js").catch(() => {});
  return THREE;
}
height = 400
camera = {
  const fov = 45;
  const aspect = width / height;
  const near = 1;
  const far = 1000;
  const camera = new THREE.PerspectiveCamera(fov, aspect, near, far);
  camera.position.z = 3;
  return camera;
}
function make_sphere(wireframe = false, opacity = 1) {
  const material = new THREE.MeshStandardMaterial({
    wireframe: wireframe, 
    opacity: opacity, 
    color: 0x555555, 
    transparent: true  
  });
  const geometry = new THREE.SphereGeometry(1, 32, 32 ); 
  return new THREE.Mesh(geometry, material);
}
scene = {
  const scene = new THREE.Scene();
  scene.background = new THREE.Color(0xffffff);
  const axes = new THREE.AxesHelper(2);
  axes.scale = new THREE.Vector(2, 2, 2);
  scene.add(axes);
  scene.add(make_sphere(/*wireframe: */false, /*opacity: */0.1));
  scene.add(make_sphere(/*wireframe: */true, /*opacity: */1));
  const light = new THREE.AmbientLight(0xFFFFFF,  /*intensity=*/1);
  scene.add(light);
  return scene;
}
